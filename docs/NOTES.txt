-> CONSULTA USUARIOS COM CONNECTIONS ABERTAS NO BANCO
        SELECT
        s.session_id,
        s.login_name,
        s.host_name,
        s.program_name,
        s.status,
        s.login_time,
        DATEDIFF(MINUTE, s.login_time, GETDATE()) AS minutos_conectado
    FROM sys.dm_exec_sessions s
    WHERE s.is_user_process = 1
    ORDER BY minutos_conectado DESC;
